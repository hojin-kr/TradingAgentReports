import pandas as pd

# Define data from YFin Data API
data = {
    'Date': ['2025-10-25', '2025-10-26', '2025-10-27', '2025-10-28', '2025-10-29', '2025-10-30', '2025-10-31', '2025-11-01'],
    'ARKK': [84.44639993286134, 82.40280095585938, 81.4751999504634, 80.63059992410374, 79.74479992108922, 78.89540093710463, 77.9778009340213, 77.01109993066155],
    'close_50_sma': [None]*8 + [84.901300, 84.479400]
}

# Create DataFrame
df = pd.DataFrame(data)

# Calculate rsi
def calculate_rsi(df):
    df['delta'] = df['ARKK'].diff().fillna(0)
    up, down = (df['delta'] > 0), (df['delta'] < 0)
    df['up_diff'] = df.up * df.delta
    df['down_diff'] = df.down * -1 * df.delta
    df['sum_up'] = df['up_diff'].cumsum()
    df['sum_down'] = (-1) * df['down_diff'].cumsum()
    df['rs'] = (df['sum_up'] / df['sum_down']).fillna(0)
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    return df

# Calculate macd
def calculate_macd(df):
    df['ema_26'] = df['ARKK'].ewm(span=26, adjust=False).mean()
    df['ema_12'] = df['ARKK'].ewm(span=12, adjust=False).mean()
    df['macd'] = df['ema_26'] - df['ema_12']
    return df

# Calculate macds
def calculate_macds(df):
    df['ema_26'] = df['ARKK'].ewm(span=26, adjust=False).mean()
    df['ema_12'] = df['ARKK'].ewm(span=12, adjust=False).mean()
    df['macds'] = df['ema_26'] - df['ema_12']
    return df

# Calculate macdh
def calculate_macdh(df):
    df['ema_26'] = df['ARKK'].ewm(span=26, adjust=False).mean()
    df['ema_12'] = df['ARKK'].ewm(span=12, adjust=False).mean()
    df['macdh'] = df['ema_26'] + df['ema_12']
    return df

# Calculate bollinger bands
def calculate_bollinger_band(df):
    df['sigma'] = df['close_50_sma'].rolling(14, min_periods=1).std() * 2.8284271247461903
    df['upper'] = df['close_50_sma'] + df['sigma']
    df['lower'] = df['close_50_sma'] - df['sigma']
    return df

# Calculate rsi
df_rsi = calculate_rsi(df)

# Calculate macd
df_macd = calculate_macd(df)

# Calculate macds
df_macds = calculate_macds(df)

df_macdh = calculate_macdh(df)
